

# Start RoboLab Docker with the VALP image:
./docker/run_docker.sh nisarga


RUN EVALUATIONS:

# Terminal 1 inside Docker: load VALP once and keep this running:
/workspace/isaaclab/_isaac_sim/python.sh valp/inference/serve_policy.py \
    --cfg-file droid-224px-8f-dual.yaml \
    --host 0.0.0.0 \
    --port 8000

# Terminal 2 inside the same Docker container: run/rerun eval:
PROTOCOL_BUFFERS_PYTHON_IMPLEMENTATION=python \
    /workspace/isaaclab/_isaac_sim/python.sh examples/policy/run_eval.py \
    --policy valp \
    --num-runs 3 \
    --num-envs 1 \
    --task ReachBananaTask \
    --remote-host localhost \
    --remote-port 8000

# All model versions:
# Stop any manually started VALP server first. This script starts each cfg,
# runs 30 eval episodes, summarizes, kills that server, then moves to the next cfg.
# Output folders are created by run_eval.py from hosted model metadata, e.g.
# robolab/output/dual_dinov3, robolab/output/ind_vjepa, etc.
# After each model, the sweep writes robolab/output/<modelname>_success_summary.txt,
# zips robolab/output/<modelname>.zip, and removes the expanded output folder.
HEADLESS=1 VIDEO_MODE=sensor bash examples/policy/run_reach_valp_eval_5x.sh
HEADLESS=1 VIDEO_MODE=sensor bash examples/policy/run_reach_valp_model_sweep_5x.sh
HEADLESS=1 VIDEO_MODE=sensor bash examples/policy/run_reach_valp_model_sweep_224_5x.sh
HEADLESS=1 VIDEO_MODE=sensor bash examples/policy/run_reach_valp_model_sweep_256_5x.sh
